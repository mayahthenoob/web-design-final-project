<!-- JavaScript -->
<script>
function filterList() {
  const input = document.getElementById("myInput").value.toUpperCase();
  const ul = document.getElementById("myUL");
  const li = ul.getElementsByTagName("li");

  for (let i = 0; i < li.length; i++) {
    const a = li[i].getElementsByTagName("a")[0];
    const txtValue = a.textContent || a.innerText;
    li[i].style.display = txtValue.toUpperCase().indexOf(input) > -1 ? "" : "none";
  }

  // Clear active letter filter if manual search is used
  clearActiveLetters();
}

function sortList(ascending = true) {
  const ul = document.getElementById("myUL");
  const items = Array.from(ul.getElementsByTagName("li"));

  items.sort((a, b) => {
    const textA = a.textContent.trim().toLowerCase();
    const textB = b.textContent.trim().toLowerCase();
    if (textA < textB) return ascending ? -1 : 1;
    if (textA > textB) return ascending ? 1 : -1;
    return 0;
  });

  // Re-append sorted items
  items.forEach(item => ul.appendChild(item));
}

// A-Z Filter Setup
function setupAlphabetFilter() {
  const container = document.getElementById("alphabetFilter");
  const alphabet = "ABCDEFGHIJKLMNOPQRSTUVWXYZ".split("");

  alphabet.forEach(letter => {
    const btn = document.createElement("button");
    btn.textContent = letter;
    btn.onclick = () => filterByLetter(letter, btn);
    container.appendChild(btn);
  });

  // Add "All" button to reset filter
  const allBtn = document.createElement("button");
  allBtn.textContent = "All";
  allBtn.onclick = () => {
    showAll();
    clearActiveLetters();
  };
  container.appendChild(allBtn);
}

function filterByLetter(letter, btn) {
  const ul = document.getElementById("myUL");
  const li = ul.getElementsByTagName("li");

  for (let i = 0; i < li.length; i++) {
    const a = li[i].getElementsByTagName("a")[0];
    const txtValue = a.textContent || a.innerText;
    li[i].style.display = txtValue.toUpperCase().startsWith(letter) ? "" : "none";
  }

  clearActiveLetters();
  btn.classList.add("active");

  // Clear the search box
  document.getElementById("myInput").value = "";
}

function clearActiveLetters() {
  const buttons = document.querySelectorAll(".alphabet-filter button");
  buttons.forEach(btn => btn.classList.remove("active"));
}

function showAll() {
  const ul = document.getElementById("myUL");
  const li = ul.getElementsByTagName("li");

  for (let i = 0; i < li.length; i++) {
    li[i].style.display = "";
  }

  // Clear the search box
  document.getElementById("myInput").value = "";
}

// Initialize A-Z filter on page load
setupAlphabetFilter();
</script>
